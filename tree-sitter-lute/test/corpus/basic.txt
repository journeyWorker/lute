==================
match / when / otherwise (nesting)
==================
<match on="scene.choices.number">
  <when test="@fond">
    :fixer{delivery="thought"}: Asked nicely.
  </when>
  <when test="$ == 'blunt'">
    :fixer{delivery="thought"}: Asked flatly.
  </when>
  <otherwise>
    :fixer{delivery="thought"}: Somewhere this evening.
  </otherwise>
</match>
---
(source_file
  (match
    (cel_attr (cel_key) (cel_string (path)))
    (when (cel_attr (cel_key) (cel_string (ref))) (line (speaker) (attrs (attr (key) (string))) (text)))
    (when (cel_attr (cel_key) (cel_string)) (line (speaker) (attrs (attr (key) (string))) (text)))
    (otherwise (line (speaker) (attrs (attr (key) (string))) (text)))))

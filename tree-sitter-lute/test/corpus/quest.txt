==================
quest with objective and on (nesting parity with branch/match)
==================
<quest id="rescueHalsin">
  <objective id="reachGrove" done="run.region == 'grove'">
    :line[narrator]: You made it to the grove.
  </objective>
  <on event="questComplete">
    ::set{user.xp += 300}
  </on>
</quest>
---
(source_file
  (quest
    (attr (key) (string))
    (objective
      (attr (key) (string))
      (cel_attr (cel_key) (cel_string (path)))
      (line (speaker) (text)))
    (on
      (attr (key) (string))
      (set (path) (assign_op) (cel_expr)))))

    ::set{user.xp += 1}
  </on>
</quest>
